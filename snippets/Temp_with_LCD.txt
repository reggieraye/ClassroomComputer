#include <Wire.h>
#include "rgb_lcd.h"
#include <math.h>

const int B = 4275;        // B value of the thermistor
const int R0 = 100000;     // R0 = 100k
const int pinTempSensor = A0; // Grove -Temperature Sensor connect to A0

rgb_lcd lcd;

void setup() {
    // set up the LCD's number of columns and rows:
    lcd.begin(16, 2);
    lcd.setRGB(255, 255, 255); // Default color before reading temperature
    analogWrite(10, 128);
}

void loop() {
    int a = analogRead(pinTempSensor);
    float R = 1023.0 / a - 1.0;
    R = R0 * R;
    float temperatureC = 1.0 / (log(R / R0) / B + 1 / 298.15) - 273.15;
    float temperatureF = (temperatureC * 9.0 / 5.0) + 32.0;

    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("Fahrenheit:");
    lcd.setCursor(0, 1);
    lcd.print(temperatureF, 1);

    // Change the background color based on temperature
    if (temperatureF <= 68) {
        lcd.setRGB(0, 0, 255); // Blue
    } else if (temperatureF > 68 && temperatureF <= 71) {
        lcd.setRGB(0, 255, 0); // Green
    } else {
        lcd.setRGB(255, 0, 0); // Red
    }

    delay(500);
}
